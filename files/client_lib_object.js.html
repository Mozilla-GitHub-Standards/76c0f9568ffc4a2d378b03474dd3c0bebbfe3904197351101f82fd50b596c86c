/**
 * Helper functions for working with Objects
 */
define([], function () {
  'use strict';

  function extend(target) {
    var sources = [].slice.call(arguments, 1);

    for (var index = 0, source; source = sources[index]; ++index) {
      for (var key in source) {
        target[key] = source[key];
      }
    }

    return target;
  }

  return {
    extend: extend
  };
});
